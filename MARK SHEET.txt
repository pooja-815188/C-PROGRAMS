int checkprime(int n){
int flag=0;
if(n==0||n==1){
flag=1;
}
else{
for(int i=2;i<n;i++){
if(n%i==0){
flag=1;
break;
}
}
}
if(flag!=1){
return 1;
}
else
return 0;
}
